#java.util.logging format
handlers = jmxlogger.integration.logutil.JmxLoggingHander
jmx.logger.Handler.level=ALL|SEVERE|INFO|FINE|etc
jmx.logger.Handler.formatter = java.util.logging.XMLFormatter|any standard formatter
jmx.logger.Handler.objectName=JMX Object Name
jmx.logger.Handler.mbeanServer="platform"|domain name of server
jmx.logger.filterExpression=#{log.level == "INFO" && log.matches("/pattern/")}
jmx.logger.filterScript=file://some/local/file.mvl
jmx.logger.filter=java.util.logging.Filter


#log4j xml format
<log4j:configuration xmlns:log4j="http://jakarta.apache.org/log4j/">
  <appender name="jmxlogger" class="jmxlogger.integration.log4j.JmxLogAppender">
    <param name="ObjectName" value="log4j.logging:type=Log4jAppender"/>
    <param name="MBeanServer" value="platform"/>
    <param name="FilterExpression" value="#{log.level == "INFO" && log.matches("/pattern/")}"/>
    <param name="FilterScript" value="file://some/local/file.mvl"/>

    <layout class="org.apache.log4j.PatternLayout">
      <param name="ConversionPattern" value="%-5p %c{1} - %m%n"/>
    </layout>
    <filter class="jmxlogger.integration.log4j.LogFilter"/>
  </appender>
</log4j:configuration>
